						    int buscaSequencial(int array[], int n, int m) {
							    int i;
							    for(i = 0; i < n; i++) {
								    if(array[i] == m) {
									    return i;
								    }
							    }
							    return -1;
						    }
					    
				    
    				        
						    int buscaSequencialRecursiva(int array[], int n, int m) {
							    if(n == 0)
								    return -1;
							    else if(array[n-1] == m)
								    return n-1;
							    else
								    return buscaSequencialRecursiva(array, n-1, m);
						    }
					    
				    
    				        
						    int buscaSequencialOrdenada(int array[], int n, int m) {
							    int i;
							    for(i = 0; i < n; i++) {
								    if(array[i] == m)
									    return i;
								    else if(array[i] > m)
									    return -1;
							    }
							    return -1;
						    }
					    
				    
    				        
						    int buscaSequencialOrdenadaRecursiva(int array[], int n, int m) {
							    if(n == 0 || array[n-1] > m)
								    return -1;
							    else if(array[n-1] == m)
								    return n-1;
							    else
								    return buscaSequencialOrdenadaRecursiva(array, n-1, m);
						    }
					    
				    
    				        
						    int buscaBinaria(int array[], int n, int m) {
							    int first, middle, last;
							    first = 0; last = n-1;
							    while (first <= last) {
								    middle = (first + last)/2;
								    if (array[middle] == m) return middle;
								    if (array[middle] < m) first = middle + 1;
								    else last = middle - 1;
							    }
							    return -1;
						    }
					    
				    
    				        
						    int buscaBinariaRecursiva (int array, first, last, m) {
						        if (first > last) return -1;
						        else {
						            int middle = (first + last)/2;
						            if (array[middle] == m) return middle;
						            if (array[middle] < m)
						                return buscaBinariaRecursiva (array, middle+1, last, m);
						            else
						                return buscaBinariaRecursiva (array, first, middle-1, m);
						        }
						    }
